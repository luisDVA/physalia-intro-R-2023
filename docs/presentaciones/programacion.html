# obtener min, max, y la diferencia
resumenMMR <- function(x){
  min_x <- min(x,na.rm = TRUE)
  max_x <- max(x,na.rm = TRUE)
  maxmin <- max_x - min_x
  c(min_x, max_x, maxmin)
}
